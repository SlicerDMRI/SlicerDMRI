36 reader->SetFileName(inputVolume.c_str() );
37 reader->Update();

46 vtkNew<vtkMRMLNRRDStorageNode> helper;
47
48 if( !helper->ParseDiffusionInformation(reader.GetPointer(), grads.GetPointer(), bValues.GetPointer()) )
49 {
50 std::cerr << argv[0] << ": Error parsing Diffusion information" << std::endl;
51 return EXIT_FAILURE;
52 }
53 vtkNew<vtkTeemEstimateDiffusionTensor> estim;
54
55 estim->SetInputConnection(reader->GetOutputPort() );
56 estim->SetNumberOfGradients(grads->GetNumberOfTuples() );
57 estim->SetDiffusionGradients(grads.GetPointer());
58 estim->SetBValues(bValues.GetPointer());
59 estim->SetShiftNegativeEigenvalues(ShiftNegativeEigenvalues);
60
61 // Compute Transformation that brings the gradients to ijk
62 // double *sp = reader->GetOutput()->GetSpacing();
63 vtkNew<vtkMatrix4x4> mf;
64 mf->DeepCopy(reader->GetMeasurementFrameMatrix() );
65 vtkNew<vtkMatrix4x4> rasToIjkRotation;
66 rasToIjkRotation->DeepCopy(reader->GetRasToIjkMatrix() );
67 // Set Translation to zero
68 for( int i = 0; i < 3; i++ )
69 {
70 rasToIjkRotation->SetElement(i, 3, 0);
71 }
72 // Remove scaling in rasToIjk to make a real rotation matrix
73 double col[3];
74 for( int jjj = 0; jjj < 3; jjj++ )
75 {
76 for( int iii = 0; iii < 3; iii++ )
77 {
78 col[iii] = rasToIjkRotation->GetElement(iii, jjj);
79 }
80 vtkMath::Normalize(col);
81 for( int iii = 0; iii < 3; iii++ )
82 {
83 rasToIjkRotation->SetElement(iii, jjj, col[iii]);
84 }
85 }
86
87 vtkNew<vtkTransform> trans;
88 trans->PostMultiply();
89 trans->SetMatrix(mf.GetPointer());
90 trans->Concatenate(rasToIjkRotation.GetPointer());
91 trans->Update();
92
93 estim->SetTransform(trans.GetPointer());
94 if( estimationMethod == std::string("LS") )
95 {
96 estim->SetEstimationMethodToLLS();
97 }
98 else if( estimationMethod == std::string("WLS") )
99 {
100 estim->SetEstimationMethodToWLS();

102 estim->Update();
103 vtkImageData *tensorImage = estim->GetOutput();
104 // Read the tensor mask

106 if( strlen(inputMaskVolume.c_str() ) > 0 )

109 maskReader->SetFileName(inputMaskVolume.c_str() );

126 cast->SetOutputScalarTypeToUnsignedChar();

131 }

139 tensorMask->SetNumberOfThreads(1);
140 if( applyMask )
141 {
142 tensorMask->SetMaskAlpha(0.0);
143 tensorMask->SetInputConnection(estim->GetOutputPort());
144 tensorMask->SetMaskInputData(mask.GetPointer());
145 tensorMask->Update();
146 tensorImage = tensorMask->GetOutput();
147 }
148
149 // Compute IjkToRas (used by Writer)
150 vtkMatrix4x4* ijkToRasMatrix = reader->GetRasToIjkMatrix();
151 ijkToRasMatrix->Invert();
152
153 // Don't save the scalars array, only the tensor array.
154 // Save tensor
155 vtkNew<vtkTeemNRRDWriter> writer;
156 tensorImage->GetPointData()->SetScalars(NULL);
157 writer->SetInputData(tensorImage);
158 writer->SetFileName( outputTensor.c_str() );
159 writer->UseCompressionOn();
160 writer->SetIJKToRASMatrix( ijkToRasMatrix );
161 // Compute measurement frame: Take into account that we have transformed
162 // the gradients so tensor components are defined in ijk.
163 rasToIjkRotation->Invert();
164 writer->SetMeasurementFrameMatrix(rasToIjkRotation.GetPointer());
165 writer->Write();
166
167 // Save baseline
168 vtkNew<vtkTeemNRRDWriter> writer2;
169 writer2->SetInputData(estim->GetBaseline() );
170 writer2->SetFileName( outputBaseline.c_str() );
171 writer2->UseCompressionOn();
172 writer2->SetIJKToRASMatrix( ijkToRasMatrix );
173 writer2->Write();
174
175 return EXIT_SUCCESS;
176 }

179bool transformsNotEqual(const vtkMatrix4x4* mat1, const vtkMatrix4x4* mat2)
180{
181 for( int i = 0; i < 4; i++ )
182 {
183 for( int j = 0; j < 4; j++ )
184 {
185 if( fabs(mat1->GetElement(i, j) - mat2->GetElement(i, j) ) > MAT_TOL )
186 {
187 return true;
188 }
189 }
190 }
191 return false;
